\subsection*{Ejemplos}

    \item Entrada: \(a = [1, 3, 5, 7], \, b = [2, 4, 6, 8]\)\\
    Salida: \([1, 2, 3, 4, 5, 6, 7, 8]\)\\

    \item Entrada: \(a = [10, 20, 30], \, b = []\)\\
    Salida: \([10, 20, 30]\)\\

    \item Entrada: \(a = [1, 1, 3], \, b = [1, 2, 2, 3]\)\\
    Salida: \([1, 1, 1, 2, 2, 3, 3]\)\\
\end{itemize}
