int b = int.Parse(Console.ReadLine() ?? string.Empty);

if (b != 0 && a % b == 0)
{
    Console.WriteLine($"{a} es divisible por {b}");
}
else
{
    Console.WriteLine($"{a} no es divisible por {b}");
}
