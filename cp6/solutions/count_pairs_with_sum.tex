public static int CountPairsWithSum(int[] values, int k)
{
    // Validar el array.
    if (values == null || values.Length < 2)
        return 0;

    int count = 0;

    // Recorremos el array.
    for (int i = 0; i < values.Length; i++)
    {
        for (int j = i + 1; j < values.Length; j++)
        {
            // Verificamos si el par suma k.
            if (values[i] + values[j] == k)
            {
                count++;
            }
        }
    }

    return count;
}
